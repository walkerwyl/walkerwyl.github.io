## Linux 下的vim配置

```vimscript
"=>GUI Settings-------------------------------------{{{
set guifont=Monospace\ 14
set guioptions-=mTb
"}}}
"=>Basic Settings-----------------------------------{{{
" 显示行号（当终端足够宽时）
if &co > 80
    set nu
endif
" 显示不可见字符（制表符、换行符等）
set list
" 设置命令行高度为2
set ch=2
" 总显示状态栏
set laststatus=2
" 状态栏中显示最后执行的命令
set showcmd
" 高亮显示光标所在行
set cursorline
" 显示尺度信息（右下角显示行、列、百分位）
set ruler
" 较长的文本行的漂亮显示
set display+=lastline
" 不备份文件
set nobackup
" 设置折叠模式
set fdm=marker
" 禁用方向键
map <up> <nop>
map <down> <nop>
map <left> <nop>
map <right> <nop>
"}}}
"=>Custom Keybindings------------------------------{{{
" 设置先导键为空格
" let mapleader=','
let mapleader=' '
" 自定义快捷键
" 快捷新建空白文件
noremap <leader>n :enew<cr>
" 快捷保存当前文件
noremap <leader>w :w<cr>
" 快捷展示当前文件列表
noremap <leader>f :files<cr>
" 快捷使用系统剪贴板
noremap <leader>y "+y<cr>
noremap <leader>p "+p<cr>
" 快捷切换窗口
noremap <leader>h <c-w>h
noremap <leader>j <c-w>j
noremap <leader>k <c-w>k
noremap <leader>l <c-w>l
"}}}
```
